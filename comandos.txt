ls -la : vizualiza pastas ocultas
touch arquivo.txt : cria um arquivo
ls: vizualiza os arquivos dentro do diretorio
git log --stat : estatistica de cada commit
git log --pretty=format: "%h - %an, %ar : %s" : formatando a forma d mpstrar o log
git log --since=2.days : mostra o log de commits dos ultimos dois dias
mkdir .idea : cria uma pasta chamada .idea
touch .idea/teste : cria um arquivo chamado teste dentro da pasta .idea
vim .gitignore : cria o git ignore 
git checkout '1309e0c050' : volta para commit citado
git checkout master : volta para master
git reset HEAD~1 --soft :volta um commit anterior (~2 volta dois commits), deixando pronto para ser  commitado
git reset HEAD~1 --hard : tira o ultimo commit
git checkout -b funcionalidade1 : cria uma branch com o nome funcionalidade1
ls: lista o que tem 


git merge funcionalidade1 : coloca tudo que esta na banch funcionalidade1 na branch master, criando um novo commit
git rebase funcionalidade1 : organiza os commits em uma ordem linear, traz o commit do funcionalidade1 pro master sem criar um novo commit

rm -rf pasta : apaga o diretorio
git clone https://github.com/andressadipaula/CursoGit.git Bola : clona para uma pasta chamada bola
git branch -a : vizualiza as branch remoto que existe
git checkout -b funcionalidade1 origin/funcionalidade1 : cria a branch funcionalidade1 local e traz pra o diretorio
git pull : atualiza (inclusive branch que ainda nao foi clonada no diretorio), branch q esta remoto
